function procesarExcel(workbook){ let frentes = []; let plan = []; let real = []; workbook.SheetNames.forEach(sheetName => { const sheet = workbook.Sheets[sheetName]; // ✅ Leer celdas exactas let planCell = sheet["N3"]; let realCell = sheet["R3"]; let planVal = planCell ? limpiar(planCell.v) : null; let realVal = realCell ? limpiar(realCell.v) : null; // ✅ Solo si son válidos if(planVal !== null && realVal !== null){ frentes.push(sheetName); plan.push(planVal); real.push(realVal); } }); // ✅ Validación if(frentes.length === 0){ alert("⚠️ No se encontraron datos en N3 / R3"); return; } actualizar(frentes, plan, real); }
